Thymeleaf3

th:text 文本显示

//Java代码
@RequestMapping("/index")
public String index(ModelMap model) {
    model.put("name","小学生");

    return "index";
}

//模板
<span th:text="${name}"></span>

//渲染结果
<span>小学生</span>

//模板
<span th:text="${'你好，' + name}"></span>

//渲染结果
<span>你好，小学生</span>

//${} 内部支持 四位运算 结合律 等乱七八糟的东西
<span th:text="${ 1 + 2 * 3 + 4}"></span>

分支结构 th:if th:switch

//java代码
model.put("condition",true);

//模板
<p th:if="${condition}">condition = true</p>

//渲染结果
<p>condition = true</p>

//模板
<p th:if="${1 gt 3 }></p>  // false  大于
<p th:if="${1 le 3 }></p>  // true   小于
<p th:if="${1 eq 3 }></p>  // false  等于
<p th:if="${1 ge 3 }></p>  // false  大于等于
<p th:if="${1 le 3 }></p>  // true   小于等于

<div th:if="${user.isAdmin()} == false"> //true or false
<div th:if="${user.isAdmin()} == null"> //为空判断

//th:switch
 <div th:switch="${name}">
    <p th:case="'小学生'">KS 1</p>
    <p th:case="'小伙子'">KS 2</p>
    <p th:case="*"> 默认值 value </p>
 </div>
//结果
<div>
    <p>KS 1</p>
</div>

th:each 循环结构

//Java代码
class Student {
    private int id;
    private String name;
    
    public Student() {}
    public Student(int id,String name) { ... }
    
    //get set ...   
}

@RequestMapping("/index")
public String index(ModelMap model) {
    List<Student> students = Arrays.asList(
        new Student(1,"小明"),
        new Student(2,"小红"),
        new Student(3,"小芳"));
        
     model.put("students",students);

    return "index";
}

//模板 - 1 
<ul>
    <li th:each="stu: ${students}">
        <span th:text=${stu.id}></span>
        <span th:text=${stu.name}></span>
    </li>
</ul>

//结果 - 1
<ul>
    <li>
        <span>1</span>
        <span>小明</span>
    </li>
    <li>
        <span>2</span>
        <span>小红</span>
    </li>
    <li>
        <span>3</span>
        <span>小芳</span>
    </li>
</ul>

//模板 - 2
<li th:each="stu,status: ${students}">
    <span th:text="${status.index}"></span>//当前第几个从0开始
    <span th:text="${status.count}"></span>//当前第几个从1开始
    <span th:text="${status.size}"></span> //总条数
    <span th:text="${status.odd}"></span>  //是否基数
    <span th:text="${status.even}"></span> //是否偶数
</li>

//结果 - 2
<li>
    <span>0</span>
    <span>1</span>
    <span>3</span>
    <span>true</span>
    <span>false</span>
</li>
<li>

<li>
    <span>2</span>

@{ }

@{}用来存放静态的数据，通常用在 如 script,img,link,a等标签上，如：

    //模板
    <link href="main.css" th:href="@{style.css}">
    
    //渲染后结果为
    <link href="style.css">
    
    //当然 在静态数据中载入带动数据也是允许的 如 当 name='small' 时
    <link  th:href="@{'style/' + '${name}' + '/min.css'}">
    //渲染后的结果为:
    <link href="style/small/min.css">
    
    //更多用法去官网

*{ }

当使用 th:object 存放对象时候，它的下级就可以使用*{}直接读取对象中属性如：

//java 代码
model.put("student",new Student(1,"小明"));

//模板
<div th:object="${student}">
    <p th:text="*{id}"></p>   //直接读取 student 里的id
    <p th:text="*{name}"></p> //直接读取 student 里的name
</div>  

//结果
<div>
    <p>1</p>    
    <p>小明</p> 
</div>

有的时候我们想对象里值不为null的时候才显示否则就显示默认值，这时候可以使用?:操作符 如：<span th:text="*{name}?: '这里是默认值' ">

无操作符 _

直接例子：

//如果name的值为null 则显示 标签里的值 否则显示name值
<span th:text="${name} ?: _">标签里的值</span>

Template Layout th:fragment

如 定义公共页头 页脚

<!-- footer.html 使用th:fragement 定义页脚的名字-->
<!DOCTYPE html>
<html xmlns:th="http://www.thymeleaf.org">
    <body>
        <div th:fragment="copy">
            &copy; 2011 The Good Thymes Virtual Grocery
        </div>
    </body>
</html>

<!-- 其他页面引入页脚如下 -->
<html xmlns:th="http://www.thymeleaf.org">
    <head>....</head>
    <body>
        ....
        ....
        <!-- 使用 ~{文件名 :: 片段名} 插入公共页脚模板 -->
        <!-- th:insert 保留当前外部div  -->
        <!-- th:replace 不保留当前外层div -->
        <div th:insert="~{footer :: copy}"></div>
    </body>
</html>


<!-- header.html 定义可以才参数的页头 -->
<!DOCTYPE html>
<html xmlns:th="http://www.thymeleaf.org">
<head th:fragment="common_header(title,links)">
    <!-- 这里可以根据传入的title 替换当前的标题 -->
    <title th:replace="${title}">这是公共的标题</title>
    
    <link rel="stylesheet" type="text/css"  th:href="@{/css/common.css}">
    <script th:src="@{/sh/scripts/codebase.js}"></script>

    <!-- 这里可以根据传入的links 自定义添加css引用 -->
    <th:block th:replace="${links}" />
</head>
</html>

<!-- 其他页面引入页头如下 -->
<!DOCTYPE html>
<html xmlns:th="http://www.thymeleaf.org">
<head th:replace="header::common_header(~{::title},~{::link})">
    <title>用户页面标题</title>
    <link href="style.css" th:href="@{'static/user.css'}">
</head>
<body>
    ....
</body>
</html>


<!-- 渲染后结果如下 -->

<!DOCTYPE html>
<html>
<head>
    <title>用户页面标题</title>
    <link rel="stylesheet" type="text/css" href="/css/common.css">
    <script src="/sh/scripts/codebase.js"></script>

    <link href="static/user.css">
</head>
<body>
    ....
</body>
</html>

<!-- 如果不需要 添加额外的 link 可以使用 -->
<head th:replace="header::common_header(~{::title},~{})">

<!-- 如果采用的内部的标题 可以使用 -->
<head th:replace="header::common_header(_,~{})">

<!-- 根据条件引入片段-->
<div th:insert="${user.isAdmin()} ? ~{common :: adminhead} : ~{}">...</div>
